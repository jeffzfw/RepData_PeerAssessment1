2.Devise a strategy for filling in all of the missing values in the dataset. fill the NA with average value for that 5-min interval

3.Create a new dataset that is equal to the original dataset but with the missing data filled in.

datanew <- dt
for(i in 1:nrow(datanew)){
  if(is.na(datanew$steps[i])){
    datanew$steps[i] <- stepsn$steps[which(datanew$interval[i]==stepsn$interval)]
  }
}
datanew <- arrange(datanew,interval)
